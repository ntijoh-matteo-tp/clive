def send()
  @session.print "HTTP/1.1 #{@status}\r\n"
  @session.print "Content-Type: #{@content_type}\r\n"
  @session.print "Content-Length: #{@content_length}\r\n"
  @session.print "\r\n"
  @session.print @content
  @session.close
end
